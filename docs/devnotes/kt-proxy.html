tic = TIClient.where("name like ?", '%05%').first
KTProxies generieren
conn.card_terminals.each do |ct|
  attribs = Cocard::NewKTProxy.new(card_terminal: ct).attributes
  tic.kt_proxies.create(attribs.merge(ti_client_id: tic.id))
end
