CA aufsetzen

mkdir /etc/ssl/telematik
cd /etc/ssl/telematik
mkdir certs crl newcerts private reqs
chmod 700 private
touch index.txt
echo 1000 > serial
cp /etc/ssl/openssl.cnf /etc/ssl/telematik/.
